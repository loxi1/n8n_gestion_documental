def should_use_qr(fields: dict) -> bool:
    tipo = fields.get("tipo_documental")

    if tipo not in ("factura", "guia_remision"):
        return False

    # campos críticos
    if not fields.get("serie"):
        return True
    if not fields.get("numero"):
        return True
    if not fields.get("ruc"):
        return True
    if not fields.get("fecha_emision"):
        return True

    return False

def _decode_qr_from_ndarray(img):
    if img is None:
        return []

    if not hasattr(img, "shape"):
        return []

    if img.size == 0:
        return []

    try:
        detector = cv2.QRCodeDetector()

        data, _, _ = detector.detectAndDecode(img)

        results = []
        if data:
            results.append(data.strip())

        try:
            ok, decoded_info, _, _ = detector.detectAndDecodeMulti(img)
            if ok:
                for d in decoded_info:
                    if d:
                        d = d.strip()
                        if d not in results:
                            results.append(d)
        except:
            pass

        return results

    except Exception:
        return []

def _get_qr_crops(img):
    h, w = img.shape[:2]

    return [
        img[int(h*0.60):h, int(w*0.50):w],  # bottom-right
        img[int(h*0.60):h, 0:int(w*0.50)],  # bottom-left
        img[:, int(w*0.70):w],              # right-full
        img[int(h*0.40):h, :],              # bottom-full
    ]

def is_documento_valido_produccion(fields, qr_data):
    tipo = fields.get("tipo_documental")

    if tipo == "factura":
        return (
            fields.get("serie")
            and fields.get("numero")
            and fields.get("ruc")
            and fields.get("fecha_emision"),
            [],
            [],
        )

    if tipo == "guia_remision":
        return (
            fields.get("serie")
            and fields.get("numero")
            and fields.get("ruc"),
            [],
            [],
        )

    if tipo == "orden_compra":
        return (
            fields.get("numero") is not None,
            [],
            [],
        )

    return False, [], []
