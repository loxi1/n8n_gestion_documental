from pypdf import PdfReader

files = [
    "/mnt/c/D/Proyectos/GestionDocumental/n8n-local/storage/files/ocr_tmp/2026/04/ocr_posible_documento_20260416204138431_1.pdf",
    "/mnt/c/D/Proyectos/GestionDocumental/n8n-local/storage/files/ocr_tmp/2026/04/ocr_posible_documento_20260416204138431_3.pdf",
    "/mnt/c/D/Proyectos/GestionDocumental/n8n-local/storage/files/ocr_tmp/2026/04/ocr_posible_documento_20260416204138431_5.pdf",
]

for path in files:
    print("=" * 80)
    print(path)
    pdf = PdfReader(path)
    text = "\n".join((p.extract_text() or "") for p in pdf.pages)
    print(text[:2000])
    print()
