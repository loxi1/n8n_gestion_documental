def _extract_oc_fields(text_u: str, name_u: str):
    patrones = [
        r"ORDEN\s+DE\s+COM(?:P|R)A.*?(\d{4,})",
        r"ORDEN\s+COMPRA.*?(\d{4,})",
        r"OC[- ]?(\d{4,})",
        r"N[°º*:]?\s*:?\s*(\d{4,})",
    ]

    for fuente in (text_u, name_u):
        for patron in patrones:
            m = re.search(patron, fuente, re.IGNORECASE | re.DOTALL)
            if m:
                return "OC", m.group(1)

    return None, None
👉 Este ya cubre:

formatos rotos de Starsoft
