SQL
SELECT column_name, data_type, character_maximum_length, is_nullable
FROM information_schema.columns
WHERE table_name = 'documentos'
ORDER BY ordinal_position;

SQL recomendado ahora
Tabla principal
SQL
CREATE TABLE IF NOT EXISTS clientes_destino (
    id BIGSERIAL PRIMARY KEY,
    nombre_oficial TEXT NOT NULL,
    abreviatura VARCHAR(30) NOT NULL UNIQUE,
    ruc VARCHAR(15),
    ruta_windows TEXT NOT NULL,
    descripcion TEXT,
    estado BOOLEAN NOT NULL DEFAULT TRUE,
    creado_en TIMESTAMP NOT NULL DEFAULT NOW(),
    actualizado_en TIMESTAMP NOT NULL DEFAULT NOW()
);
Alias
SQL
CREATE TABLE IF NOT EXISTS clientes_destino_alias (
    id BIGSERIAL PRIMARY KEY,
    cliente_destino_id BIGINT NOT NULL REFERENCES clientes_destino(id),
    alias TEXT NOT NULL,
    estado BOOLEAN NOT NULL DEFAULT TRUE,
    creado_en TIMESTAMP NOT NULL DEFAULT NOW()
);
