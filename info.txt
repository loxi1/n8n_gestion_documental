7. Lógica práctica de movimiento
Función ejemplo

def construir_ruta_windows(cliente_destino: str, fecha_emision: date) -> str:
    config = CLIENTES_DESTINO.get(cliente_destino)
    if not config:
        raise ValueError(f"Cliente destino no configurado: {cliente_destino}")

    mes = MESES_ES[fecha_emision.month]
    return rf"{config['carpeta']}\{mes}"
