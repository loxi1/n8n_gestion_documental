def _iter_qr_zones(img_bgr: np.ndarray) -> list[np.ndarray]:
    h, w = img_bgr.shape[:2]

    zones = [
        img_bgr,  # página completa
        img_bgr[0:int(h * 0.40), 0:int(w * 0.40)],                  # top_left
        img_bgr[0:int(h * 0.40), int(w * 0.60):w],                  # top_right
        img_bgr[int(h * 0.55):h, 0:int(w * 0.45)],                  # bottom_left
        img_bgr[int(h * 0.55):h, int(w * 0.55):w],                  # bottom_right
        img_bgr[0:h, int(w * 0.55):w],                              # right_full
        img_bgr[0:h, 0:int(w * 0.45)],                              # left_full
        img_bgr[int(h * 0.55):h, 0:w],                              # bottom_full
        img_bgr[int(h * 0.25):int(h * 0.75), int(w * 0.25):int(w * 0.75)],  # center
    ]

    return zones
