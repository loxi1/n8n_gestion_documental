def normalize_fecha(fecha):
    m = re.match(r"(\d{1,2})/([A-Za-z]{3})\.?/(\d{4})", fecha)
    if m:
        d, mon, y = m.groups()
        mon = MESES.get(mon[:3].upper())
        if mon:
            return f"{y}-{mon}-{int(d):02d}"
    return fecha

def _extract_oc_ruc(text_u: str) -> str | None:
    m = re.search(r"R\.?U\.?C\.?.{0,20}?([0-9]{11})", text_u, re.IGNORECASE | re.DOTALL)
    if m:
        return m.group(1)

    m = re.search(r"\b([0-9]{11})\b", text_u)
    return m.group(1) if m else None
