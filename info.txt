Cómo integrarlo a tu script en Windows
Tu ocr_service.py puede quedarse igual si ejecutas Python dentro de WSL o si llamas ocrmypdf desde WSL.
Pero para pruebas rápidas, te propongo una variante temporal en Windows:
Opción simple
En vez de ejecutar ocrmypdf directo, llama a WSL desde Python:
Python
from pathlib import Path
import subprocess


def run_ocr(input_pdf: Path, output_pdf: Path) -> bool:
    output_pdf.parent.mkdir(parents=True, exist_ok=True)

    input_wsl = str(input_pdf).replace("\\", "/").replace("C:/", "/mnt/c/")
    output_wsl = str(output_pdf).replace("\\", "/").replace("C:/", "/mnt/c/")

    cmd = [
        "wsl",
        "ocrmypdf",
        "--force-ocr",
        "-l", "spa",
        input_wsl,
        output_wsl,
    ]

    try:
        result = subprocess.run(cmd, capture_output=True, text=True, check=False)
        if result.returncode != 0:
            print(result.stderr)
        return result.returncode == 0
    except FileNotFoundError:
        return False
