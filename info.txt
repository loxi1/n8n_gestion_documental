def _extract_oc_fields(text_u: str, name_u: str):
    patrones = [
        r"ORDEN\s+DE\s+COM\w{0,3}.*?([0-9]{4,})",
        r"ORDEN\s+COMPRA.*?([0-9]{4,})",
        r"OCBBTI.*?([0-9]{4,})",
        r"\bOC[:\s-]*([0-9]{4,})\b",
        r"\b([0-9]{4,})\.PDF\b",
    ]

    for fuente in (text_u, name_u):
        for patron in patrones:
            m = re.search(patron, fuente, re.IGNORECASE | re.DOTALL)
            if m:
                return "OC", m.group(1)

    return None, None
